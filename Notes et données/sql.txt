SELECT
  c.codeclasse,p.codeperiode,
  i.email,nom,prenom,codestage,
  p.datedebut,datefin
FROM periode_stage p
LEFT OUTER JOIN classe c 
ON c.codeclasse = p.codeclasse
LEFT OUTER JOIN inscrit i 
ON i.codeclasse = c.codeclasse
INNER JOIN utilisateurs u 
ON u.email = i.email
LEFT OUTER JOIN stage s 
ON s.codeperiode = p.codeperiode 
AND s.codeinscription = i.codeinscription
ORDER BY 1,p.codeperiode,nom
